const sql = require('mssql');

// 1. Identificação do Ambiente
const isAzureEnv = !!process.env.WEBSITE_SITE_NAME;
// 2. Definição do Alvo (Azure ou Local) - Se estiver na Azure, o alvo é SEMPRE azure
const dbTarget = isAzureEnv ? 'azure' : (process.env.DB_TARGET || 'local');

console.log(`📡 Alvo de Banco de Dados: ${dbTarget.toUpperCase()} ${isAzureEnv ? '(Azure App Service)' : '(Localhost)'}`);

let dbConfig;

if (dbTarget === 'local') {
  // CONFIGURAÇÃO LOCAL (Usando as variáveis do seu print)
  const serverRaw = (process.env.DB_SERVER || process.env.LOCAL_DB_SERVER || 'localhost').trim();
  const [serverAddress, instanceName] = serverRaw.split('\\');

  dbConfig = {
    server: serverAddress,
    port: parseInt(process.env.LOCAL_DB_PORT || 1433),
    user: process.env.DB_USER || process.env.LOCAL_DB_USER,
    password: process.env.DB_PASSWORD || process.env.LOCAL_DB_PASSWORD,
    database: process.env.LOCAL_DB_NAME || 'gestaointernabi-database',
    options: {
      encrypt: false,
      trustServerCertificate: true,
      useUTC: false,
      connectTimeout: 30000
    }
  };

  // Se houver uma instância (ex: SQLEXPRESS01), adicionamos ela
  if (instanceName) {
    dbConfig.options.instanceName = instanceName;
  }

  // Autenticação Inteligente (Windows ou SQL)
  const finalUser = process.env.DB_USER || process.env.LOCAL_DB_USER;
  const finalPass = process.env.DB_PASSWORD || process.env.LOCAL_DB_PASSWORD;
  const authType = process.env.LOCAL_DB_AUTH_TYPE || (finalUser ? 'sql' : 'windows');

  if (authType === 'windows' && !finalUser) {
    dbConfig.user = undefined;
    dbConfig.password = undefined;
  } else {
    dbConfig.user = finalUser;
    dbConfig.password = finalPass;
  }

} else {
  // CONFIGURAÇÃO AZURE (Produção)
  const serverRaw = (process.env.AZURE_SQL_SERVER || process.env.DB_SERVER || '').trim();
  const [serverAddress, instanceName] = serverRaw.split('\\');

  dbConfig = {
    user: process.env.AZURE_SQL_USERNAME || process.env.DB_USER,
    password: process.env.AZURE_SQL_PASSWORD || process.env.DB_PASSWORD,
    database: process.env.AZURE_SQL_DATABASE || process.env.DB_DATABASE,
    server: serverAddress,
    port: parseInt(process.env.AZURE_SQL_PORT || process.env.DB_PORT || 1433),
    options: {
      encrypt: true,
      trustServerCertificate: false,
      instanceName: instanceName,
      useUTC: false,
      connectTimeout: 30000
    }
  };
}

const config = dbConfig;

// Validação básica para ajudar no debug do Azure
if (isAzureEnv && (!config.user || !config.password)) {
  console.warn('⚠️ Alerta: Credenciais de banco de dados não detectadas no ambiente Azure.');
}

const poolPromise = new sql.ConnectionPool(config)
  .connect()
  .then(pool => {
    console.log(`✅ Conexão SQL Server estabelecida com sucesso [${dbTarget.toUpperCase()}]`);
    return pool;
  })
  .catch(err => {
    console.error(`❌ ERRO CRÍTICO: Falha ao conectar no SQL Server [${dbTarget.toUpperCase()}]:`, err.message);
    throw err;
  });

module.exports = { poolPromise, sql, isAzure: isAzureEnv, config };
